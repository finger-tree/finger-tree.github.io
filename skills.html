Python

def egcd(a,b):
    if b==0: return (a,1,0)
    g,x1,y1 = egcd(b, a % b)
    return (g, y1, x1 - (a//b)*y1)

# example
g,x,y = egcd(240,46)
print(g,x,y)

C

#include <stdio.h>
typedef long long ll;
ll egcd(ll a,ll b,ll *x,ll *y){
  if(b==0){ *x=1; *y=0; return a; }
  ll x1,y1;
  ll g = egcd(b, a%b, &x1, &y1);
  *x = y1;
  *y = x1 - (a/b)*y1;
  return g;
}
int main(){
  ll x,y; printf("%lld\n", egcd(240,46,&x,&y)); printf("%lld %lld\n",x,y);
}

function egcd(a,b){
  if(b===0) return [a,1,0];
  const [g,x1,y1]=egcd(b,a%b);
  return [g, y1, x1 - Math.floor(a/b)*y1];
}
// example:
console.log(egcd(240,46));
